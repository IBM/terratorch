86def write_tiff(img_wrt, filename, metadata): 

87 # Adapting the number of bands to be compatible with the 

88 # output dimensions. 

89 if not is_one_band(img_wrt): 

90 count = img_wrt.shape[0] 

91 metadata["count"] = count 

92 

93 with rasterio.open(filename, "w", **metadata) as dest: 

94 if is_one_band(img_wrt): 

95 img_wrt = img_wrt[None] 

96 

97 for i in range(img_wrt.shape[0]): 

98 dest.write(img_wrt[i, :, :], i + 1) 

99 return filename

102def add_default_checkpointing_config(config): 

103 subcommand = config.get("subcommand", None) 

104 if subcommand is not None: 

105 enable_checkpointing = config[subcommand + ".trainer.enable_checkpointing"] 

106 callbacks = config[subcommand + ".trainer.callbacks"] 

107 

108 # A list for callbacks is usually expected. 

109 if callbacks: 

110 check_callbacks = [op for op in callbacks if "ModelCheckpoint" in op.class_path] 

111 

112 if len(check_callbacks) > 0: 

113 there_is_checkpointing = True 

114 else: 

115 there_is_checkpointing = False 

116 # However, if no callbacks list is provided, let's create one. 

117 else: 

118 there_is_checkpointing = False 

119 callbacks = [] 

120 

121 if enable_checkpointing: 

122 if not there_is_checkpointing: 

123 # Checkpointing is enabled, but no checkpoint rule was 

124 # provided, so let's define some default choice for it. 

125 model_checkpoint = Namespace( 

126 class_path="StateDictAwareModelCheckpoint", 

127 init_args=Namespace(filename="{epoch}", monitor="val/loss"), 

128 ) 

129 model_checkpoint_state_dict = Namespace( 

130 class_path="StateDictAwareModelCheckpoint", 

131 init_args=Namespace(filename="{epoch}_state_dict", save_weights_only=True, monitor="val/loss"), 

132 ) 

133 callbacks += [model_checkpoint, model_checkpoint_state_dict] 

134 config[subcommand + ".trainer.callbacks"] = callbacks 

135 else: 

136 logger.info( 

137 "No extra checkpoint config will be added, since the user already defined it in the callbacks." 

138 ) 

139 

140 return config

143def save_prediction( 

144 prediction, 

145 input_file_name, 

146 out_dir, 

147 dtype: str = "int16", 

148 suffix: str = "pred", 

149 output_file_name: str | None = None, 

150): 

151 mask, metadata = open_tiff(input_file_name) 

152 mask = np.where(mask == metadata["nodata"], 1, 0) 

153 mask = np.max(mask, axis=0) 

154 result = np.where(mask == 1, -1, prediction.detach().cpu()) 

155 

156 ##### Save file to disk 

157 metadata["count"] = 1 

158 metadata["dtype"] = dtype 

159 metadata["compress"] = "lzw" 

160 metadata["nodata"] = -1 

161 file_name = os.path.basename(input_file_name) 

162 

163 if not output_file_name: 

164 file_name_no_ext = os.path.splitext(file_name)[0] 

165 else: 

166 file_name_no_ext = output_file_name 

167 

168 out_file_name = file_name_no_ext + f"_{suffix}.tif" 

169 logger.info(f"Saving output to {out_file_name} ...") 

170 write_tiff(result, os.path.join(out_dir, out_file_name), metadata)

197class CustomWriter(BasePredictionWriter): 

198 """Callback class to write geospatial data to file.""" 

199 

200 def __init__(self, output_dir: str | None = None, write_interval: str = "epoch"): 

201 super().__init__(write_interval) 

202 

203 self.output_dir = output_dir

205 def write_on_batch_end(self, trainer, pl_module, prediction, batch_indices, batch, batch_idx, dataloader_idx): # noqa: ARG002 

206 # this will create N (num processes) files in `output_dir` each containing 

207 # the predictions of it's respective rank 

208 

209 # by default take self.output_dir. If None, look for one in trainer 

210 if self.output_dir is None: 

211 try: 

212 output_dir = trainer.predict_output_dir 

213 except AttributeError as err: 

214 msg = "Output directory must be passed to CustomWriter constructor or the `predict_output_dir`\ 

215 attribute must be present in the trainer." 

216 raise Exception(msg) from err 

217 else: 

218 output_dir = self.output_dir 

219 

220 # Defining a prefix for the output files 

221 if hasattr(trainer, "output_file_prefix"): 

222 output_file_prefix = trainer.output_file_prefix 

223 else: 

224 output_file_prefix = None 

225 msg = "As you don't defined a value for `--output_file_prefix` via comand line, we will use a name based on the input file names." 

226 logger.info(msg) 

227 

228 if not os.path.exists(output_dir): 

229 os.makedirs(output_dir, exist_ok=True) 

230 

231 if isinstance(prediction, torch.Tensor): 

232 filename_batch = "".join(random.choices(string.ascii_letters + string.digits, k=8)) 

233 torch.save(prediction, os.path.join(output_dir, f"{filename_batch}.pt")) 

234 

235 elif isinstance(prediction, tuple): 

236 pred_batch_, filename_batch = prediction 

237 

238 # In case of multimodal datasets, we have a dictionary 

239 # with one file per modality. We need to use a file as 

240 # input in order to copy metadata. 

241 if type(filename_batch) == dict: 

242 keys = list(filename_batch.keys()) 

243 filename_batch = filename_batch[keys[0]] 

244 

245 # If we have just a single kind of output 

246 if isinstance(pred_batch_, tuple): 

247 pred_batch, suffix = pred_batch_ 

248 

249 for prediction, file_name in zip(torch.unbind(pred_batch, dim=0), filename_batch, strict=False): 

250 save_prediction( 

251 prediction, 

252 file_name, 

253 output_dir, 

254 dtype=trainer.out_dtype, 

255 suffix=suffix, 

256 output_file_name=output_file_prefix, 

257 ) 

258 

259 # If we are outputting more than one kind of variable, as 

260 # (predictions, probabilities), for segmentation 

261 elif isinstance(pred_batch_, list): 

262 for pred in pred_batch_: 

263 pred_batch, suffix = pred 

264 

265 for p, file_name in zip(torch.unbind(pred_batch, dim=0), filename_batch, strict=False): 

266 save_prediction( 

267 p, 

268 file_name, 

269 output_dir, 

270 dtype=trainer.out_dtype, 

271 suffix=suffix, 

272 output_file_name=output_file_prefix, 

273 ) 

274 # If there is no suffix to append in the output filename 

275 elif isinstance(pred_batch_, torch.Tensor): 

276 pred_batch = pred_batch_ 

277 

278 for p, file_name in zip(torch.unbind(pred_batch, dim=0), filename_batch, strict=False): 

279 save_prediction( 

280 p, 

281 file_name, 

282 output_dir, 

283 dtype=trainer.out_dtype, 

284 output_file_name=output_file_prefix, 

285 ) 

286 else: 

287 raise ValueError( 

288 f"`pred_batch_` is expected to be in `[tuple, list, torch.Tensor]` but received {type(pred_batch_)}." 

289 ) 

290 

291 elif not prediction: 

292 logger.info("Prediction output is `None` for this case.") 

293 

294 else: 

295 raise TypeError(f"Unknown type for prediction {type(prediction)}")

320def clean_config_for_deployment_and_dump(config: dict[str, Any]): 

321 deploy_config = deepcopy(config) 

322 ## General 

323 # drop ckpt_path 

324 deploy_config.pop("ckpt_path", None) 

325 # drop checkpoints 

326 deploy_config.pop("ModelCheckpoint", None) 

327 deploy_config.pop("StateDictModelCheckpoint", None) 

328 # drop optimizer and lr sheduler 

329 deploy_config.pop("optimizer", None) 

330 deploy_config.pop("lr_scheduler", None) 

331 # drop undesirable fields, if they exist 

332 if "verbose" in deploy_config: 

333 deploy_config.pop("verbose", None) 

334 ## Trainer 

335 # remove logging 

336 deploy_config["trainer"]["logger"] = False 

337 # remove callbacks 

338 deploy_config["trainer"].pop("callbacks", None) 

339 # remove default_root_dir 

340 deploy_config["trainer"].pop("default_root_dir", None) 

341 # set mixed precision by default for inference 

342 deploy_config["trainer"]["precision"] = "16-mixed" 

343 ## Model 

344 # set pretrained to false 

345 if "model_args" in deploy_config["model"]["init_args"]: 

346 # for the new prithvi EO v2 models, alter the backbone_pretrained value only. 

347 if "pretrained" in deploy_config["model"]["init_args"]["model_args"]: 

348 deploy_config["model"]["init_args"]["model_args"]["pretrained"] = False 

349 elif "backbone_pretrained" in deploy_config["model"]["init_args"]["model_args"]: 

350 deploy_config["model"]["init_args"]["model_args"]["backbone_pretrained"] = False 

351 

352 return yaml.safe_dump(deploy_config)

355class StudioDeploySaveConfigCallback(SaveConfigCallback): 

356 """Subclass of the SaveConfigCallback to save configs. 

357 Besides saving the usual config, will also save a modified config ready to be uploaded. 

358 Handles things like disabling the pretrained flag, removing loggers, ... 

359 """ 

360 

361 def __init__( 

362 self, 

363 parser, 

364 config, 

365 config_filename: str = "config.yaml", 

366 overwrite: bool = False, 

367 multifile: bool = False, 

368 save_to_log_dir: bool = True, 

369 ): 

370 super().__init__(parser, config, config_filename, overwrite, multifile, save_to_log_dir) 

371 set_dumper("deploy_config", clean_config_for_deployment_and_dump) 

372 

373 # Preparing information to save config file to log dir 

374 config_dict = config.as_dict() 

375 self.config_path_original = str(config_dict["config"][0]) 

376 _, self.config_file_original = os.path.split(self.config_path_original) 

377 

378 self.deploy_config_file = config_dict["deploy_config_file"] 

379 

380 def setup(self, trainer: Trainer, pl_module: LightningModule, stage: str) -> None: 

381 if self.already_saved: 

382 return 

383 

384 if self.save_to_log_dir: 

385 log_dir = trainer.log_dir or trainer.default_root_dir # this broadcasts the directory 

386 if log_dir is None: 

387 msg = "log_dir was none. Please set a logger or set default_root_dir for the trainer" 

388 raise Exception(msg) 

389 config_path = os.path.join(log_dir, self.config_filename) 

390 fs = get_filesystem(log_dir) 

391 

392 if not self.overwrite: 

393 # check if the file exists on rank 0 

394 file_exists = fs.isfile(config_path) if trainer.is_global_zero else False 

395 # broadcast whether to fail to all ranks 

396 file_exists = trainer.strategy.broadcast(file_exists) 

397 if file_exists: 

398 msg = ( 

399 f"{self.__class__.__name__} expected {config_path} to NOT exist. Aborting to avoid overwriting" 

400 " results of a previous run. You can delete the previous config file," 

401 " set `LightningCLI(save_config_callback=None)` to disable config saving," 

402 ' or set `LightningCLI(save_config_kwargs={"overwrite": True})` to overwrite the config file.' 

403 ) 

404 raise RuntimeError(msg) 

405 

406 if trainer.is_global_zero: 

407 # save only on rank zero to avoid race conditions. 

408 # the `log_dir` needs to be created as we rely on the logger to do it usually 

409 # but it hasn't logged anything at this point 

410 fs.makedirs(log_dir, exist_ok=True) 

411 

412 if self.deploy_config_file: 

413 self.parser.save( 

414 self.config, config_path, skip_none=True, overwrite=self.overwrite, multifile=self.multifile 

415 ) 

416 

417 if trainer.is_global_zero: 

418 if self.deploy_config_file: 

419 # also save the config that will be deployed 

420 config_name, config_ext = os.path.splitext(self.config_filename) 

421 config_name += "_deploy" 

422 config_name += config_ext 

423 config_path = os.path.join(log_dir, config_name) 

424 self.parser.save( 

425 self.config, 

426 config_path, 

427 format="deploy_config", 

428 skip_none=True, 

429 overwrite=self.overwrite, 

430 multifile=self.multifile, 

431 ) 

432 self.already_saved = True 

433 

434 config_path_dir, config_path_file = os.path.split(config_path) 

435 self.config_path_new = os.path.join(config_path_dir, self.config_file_original) 

436 

437 # broadcast so that all ranks are in sync on future calls to .setup() 

438 self.already_saved = trainer.strategy.broadcast(self.already_saved) 

439 # Copying config file to log dir only if they are different files 

440 if os.path.abspath(self.config_path_original) != os.path.abspath(self.config_path_new): 

441 shutil.copyfile(self.config_path_original, self.config_path_new)

607 def __init__( 

608 self, 

609 trainer: Trainer, 

610 model: LightningModule, 

611 datamodule: LightningDataModule, 

612 checkpoint_path: Path | None = None, 

613 ): 

614 self.trainer = trainer 

615 self.model = model 

616 self.datamodule = datamodule 

617 if checkpoint_path: 

618 weights = torch.load(checkpoint_path, map_location="cpu", weights_only=True) 

619 if "state_dict" in weights: 

620 weights = weights["state_dict"] 

621 

622 # It removes a residual prefix (related to timm) from older 

623 # checkpoints. 

624 weights = remove_unexpected_prefix(weights) 

625 # Removing the undesirable prefix `model` from the checkpoint 

626 weights_ = {} 

627 for k, v in weights.items(): 

628 splits = k.split(".") 

629 if splits[0] == "model": 

630 splits = splits[1:] 

631 k_ = ".".join(splits) 

632 weights_[k_] = v 

633 else: 

634 weights_[k] = v 

635 

636 self.model.model.load_state_dict(weights_) 

637 

638 # dont write 

639 non_writing_callbacks = [] 

640 for callback in self.trainer.callbacks: 

641 if isinstance(callback, BasePredictionWriter): 

642 continue 

643 non_writing_callbacks.append(callback) 

644 self.trainer.callbacks = non_writing_callbacks

702 def inference_on_dir(self, data_root: Path | None = None) -> tuple[torch.Tensor, list[str]]: 

703 """Perform inference on the given data root directory 

704 

705 Args: 

706 data_root (Path): Directory with data to perform inference on 

707 Returns: 

708 A tuple with a torch tensor with all predictions and a list of corresponding input file paths 

709 

710 """ 

711 

712 if data_root: 

713 self.datamodule.predict_root = data_root 

714 predictions = self.trainer.predict(model=self.model, datamodule=self.datamodule, return_predictions=True) 

715 

716 # In some cases, the output has the format ((prediction_tensor, 

717 # prediction_name), filename) 

718 if all([type(j[0]) == tuple for j in predictions]): 

719 concat_predictions = torch.cat([batch[0][0] for batch in predictions]) 

720 else: 

721 concat_predictions = torch.cat([batch[0] for batch in predictions]) 

722 

723 concat_file_names = flatten([batch[1] for batch in predictions]) 

724 

725 return concat_predictions, concat_file_names

735 

736 with tempfile.TemporaryDirectory() as tmpdir:

738 prediction, file_name = self.inference_on_dir( 

739 tmpdir, 

740 ) 

741 return prediction.squeeze(0)

745 def compute_statistics(self, datamodule: LightningDataModule, **kwargs) -> None: 

746 """ 

747 Compute the dataset statistics for the training dataset. 

748 

749 This method will compute the mean and standard deviation of the image data and the count and percentage of each 

750 unique value in the masks in case these are int and the mean and standard deviation of the mask values in case 

751 these are floats. The statistics are computed using the entire training dataset and are printed to the logger. 

752 

753 Please note that this method assumes that there is only one train dataloader in the datamodule. The train 

754 transforms are removed before computing the statistics to ensure that the statistics are computed on the raw 

755 data without any augmentation and randomization. 

756 """ 

757 # remove train transforms, this may not work for all datamodules 

758 if hasattr(datamodule, "train_transform"): 

759 datamodule.train_transform = None 

760 datamodule.setup("fit") 

761 original_dataloader = datamodule.train_dataloader() 

762 if not isinstance(original_dataloader, DataLoader): 

763 msg = "DataLoader not found in datamodule.train_dataloader()" 

764 raise ValueError(msg) 

765 new_dataloader = DataLoader( 

766 dataset=original_dataloader.dataset, 

767 batch_size=original_dataloader.batch_size, 

768 shuffle=False, 

769 num_workers=original_dataloader.num_workers, 

770 collate_fn=original_dataloader.collate_fn, 

771 pin_memory=original_dataloader.pin_memory, 

772 drop_last=False, 

773 ) 

774 image_stats = compute_statistics(new_dataloader) 

775 logger.info("Image statistics:") 

776 logger.info(yaml.dump(image_stats)) 

777 if "mask" in datamodule.train_dataloader().dataset[0]: 

778 mask_stats = compute_mask_statistics(new_dataloader) 

779 logger.info("Mask statistics:") 

780 logger.info(yaml.dump(mask_stats))
