20class UNet(nn.Module): 

21 """UNet backbone. 

22 

23 This backbone is the implementation of `U-Net: Convolutional Networks 

24 for Biomedical Image Segmentation <https://arxiv.org/abs/1505.04597>`_. 

25 

26 Args: 

27 in_channels (int): Number of input image channels. Default" 3. 

28 out_channels (int): Number of base channels of each stage. 

29 The output channels of the first stage. Default: 64. 

30 num_stages (int): Number of stages in encoder, normally 5. Default: 5. 

31 strides (Sequence[int 1 | 2]): Strides of each stage in encoder. 

32 len(strides) is equal to num_stages. Normally the stride of the 

33 first stage in encoder is 1. If strides[i]=2, it uses stride 

34 convolution to downsample in the correspondence encoder stage. 

35 Default: (1, 1, 1, 1, 1). 

36 enc_num_convs (Sequence[int]): Number of convolutional layers in the 

37 convolution block of the correspondence encoder stage. 

38 Default: (2, 2, 2, 2, 2). 

39 dec_num_convs (Sequence[int]): Number of convolutional layers in the 

40 convolution block of the correspondence decoder stage. 

41 Default: (2, 2, 2, 2). 

42 downsamples (Sequence[int]): Whether use MaxPool to downsample the 

43 feature map after the first stage of encoder 

44 (stages: [1, num_stages)). If the correspondence encoder stage use 

45 stride convolution (strides[i]=2), it will never use MaxPool to 

46 downsample, even downsamples[i-1]=True. 

47 Default: (True, True, True, True). 

48 enc_dilations (Sequence[int]): Dilation rate of each stage in encoder. 

49 Default: (1, 1, 1, 1, 1). 

50 dec_dilations (Sequence[int]): Dilation rate of each stage in decoder. 

51 Default: (1, 1, 1, 1). 

52 with_cp (bool): Use checkpoint or not. Using checkpoint will save some 

53 memory while slowing down the training speed. Default: False. 

54 conv_cfg (dict | None): Config dict for convolution layer. 

55 Default: None. 

56 norm_cfg (dict | None): Config dict for normalization layer. 

57 Default: dict(type='BN'). 

58 act_cfg (dict | None): Config dict for activation layer in ConvModule. 

59 Default: dict(type='ReLU'). 

60 upsample_cfg (dict): The upsample config of the upsample module in 

61 decoder. Default: dict(type='InterpConv'). 

62 norm_eval (bool): Whether to set norm layers to eval mode, namely, 

63 freeze running stats (mean and var). Note: Effect on Batch Norm 

64 and its variants only. Default: False. 

65 dcn (bool): Use deformable convolution in convolutional layer or not. 

66 Default: None. 

67 plugins (dict): plugins for convolutional layers. Default: None. 

68 pretrained (str, optional): model pretrained path. Default: None 

69 init_cfg (dict or list[dict], optional): Initialization config dict. 

70 Default: None 

71 

72 Notice: 

73 The input image size should be divisible by the whole downsample rate 

74 of the encoder. More detail of the whole downsample rate can be found 

75 in `UNet._check_input_divisible`. 

76 """

98 super(UNet, self).__init__()

100 self.pretrained = pretrained 

101 assert not (init_cfg and pretrained), \ 

102 'init_cfg and pretrained cannot be setting at the same time' 

103 if isinstance(pretrained, str): 

104 warnings.warn('DeprecationWarning: pretrained is a deprecated, '

106 self.init_cfg = dict(type='Pretrained', checkpoint=pretrained) 

107 elif pretrained is None: 

108 if init_cfg is None: 

109 self.init_cfg = [ 

110 dict(type='Kaiming', layer='Conv2d'), 

111 dict( 

112 type='Constant', 

113 val=1, 

114 layer=['_BatchNorm', 'GroupNorm']) 

115 ] 

116 else: 

117 raise TypeError('pretrained must be a str or None') 

118 

119 assert dcn is None, 'Not implemented yet.' 

120 assert plugins is None, 'Not implemented yet.' 

121 assert len(strides) == num_stages, \ 

122 'The length of strides should be equal to num_stages, '\ 

123 f'while the strides is {strides}, the length of '\ 

124 f'strides is {len(strides)}, and the num_stages is '\ 

125 f'{num_stages}.' 

126 assert len(enc_num_convs) == num_stages, \ 

127 'The length of enc_num_convs should be equal to num_stages, '\ 

128 f'while the enc_num_convs is {enc_num_convs}, the length of '\ 

129 f'enc_num_convs is {len(enc_num_convs)}, and the num_stages is '\ 

130 f'{num_stages}.' 

131 assert len(dec_num_convs) == (num_stages-1), \ 

132 'The length of dec_num_convs should be equal to (num_stages-1), '\ 

133 f'while the dec_num_convs is {dec_num_convs}, the length of '\ 

134 f'dec_num_convs is {len(dec_num_convs)}, and the num_stages is '\ 

135 f'{num_stages}.' 

136 assert len(downsamples) == (num_stages-1), \ 

137 'The length of downsamples should be equal to (num_stages-1), '\ 

138 f'while the downsamples is {downsamples}, the length of '\ 

139 f'downsamples is {len(downsamples)}, and the num_stages is '\ 

140 f'{num_stages}.' 

141 assert len(enc_dilations) == num_stages, \ 

142 'The length of enc_dilations should be equal to num_stages, '\ 

143 f'while the enc_dilations is {enc_dilations}, the length of '\ 

144 f'enc_dilations is {len(enc_dilations)}, and the num_stages is '\ 

145 f'{num_stages}.' 

146 assert len(dec_dilations) == (num_stages-1), \ 

147 'The length of dec_dilations should be equal to (num_stages-1), '\ 

148 f'while the dec_dilations is {dec_dilations}, the length of '\ 

149 f'dec_dilations is {len(dec_dilations)}, and the num_stages is '\ 

150 f'{num_stages}.' 

151 self.num_stages = num_stages 

152 self.strides = strides 

153 self.downsamples = downsamples 

154 self.norm_eval = norm_eval 

155 self.out_channels = [out_channels * 2**i for i in reversed(range(num_stages))] 

156 

157 self.encoder = nn.ModuleList() 

158 self.decoder = nn.ModuleList() 

159 

160 for i in range(num_stages): 

161 enc_conv_block = [] 

162 if i != 0: 

163 if strides[i] == 1 and downsamples[i - 1]: 

164 enc_conv_block.append(nn.MaxPool2d(kernel_size=2)) 

165 upsample = (strides[i] != 1 or downsamples[i - 1]) 

166 self.decoder.append( 

167 UpConvBlock( 

168 conv_block=BasicConvBlock, 

169 in_channels=out_channels * 2**i, 

170 skip_channels=out_channels * 2**(i - 1), 

171 out_channels=out_channels * 2**(i - 1), 

172 num_convs=dec_num_convs[i - 1], 

173 stride=1, 

174 dilation=dec_dilations[i - 1], 

175 with_cp=with_cp, 

176 conv_cfg=conv_cfg, 

177 norm_cfg=norm_cfg, 

178 act_cfg=act_cfg, 

179 upsample_cfg=upsample_cfg if upsample else None, 

180 dcn=None, 

181 plugins=None)) 

182 

183 enc_conv_block.append( 

184 BasicConvBlock( 

185 in_channels=in_channels, 

186 out_channels=out_channels * 2**i, 

187 num_convs=enc_num_convs[i], 

188 stride=strides[i], 

189 dilation=enc_dilations[i], 

190 with_cp=with_cp, 

191 conv_cfg=conv_cfg, 

192 norm_cfg=norm_cfg, 

193 act_cfg=act_cfg, 

194 dcn=None, 

195 plugins=None)) 

196 self.encoder.append((nn.Sequential(*enc_conv_block))) 

197 in_channels = out_channels * 2**i

199 def forward(self, x): 

200 

201 # We can check just the first image, since the batch  

202 # already was approved by the stackability test, which means 

203 # all images has the same dimensions.  

204 self._check_input_divisible(x[0]) 

205 

206 enc_outs = [] 

207 for enc in self.encoder: 

208 x = enc(x) 

209 enc_outs.append(x) 

210 dec_outs = [x] 

211 for i in reversed(range(len(self.decoder))): 

212 x = self.decoder[i](enc_outs[i], x) 

213 dec_outs.append(x) 

214 return dec_outs

216 def train(self, mode=True): 

217 """Convert the model into training mode while keep normalization layer 

218 freezed.""" 

219 super(UNet, self).train(mode) 

220 if mode and self.norm_eval: 

221 for m in self.modules(): 

222 # trick: eval have effect on BatchNorm only 

223 if isinstance(m, _BatchNorm): 

224 m.eval()

226 def _check_input_divisible(self, x): 

227 h, w = x.shape[-2:] 

228 whole_downsample_rate = 1 

229 for i in range(1, self.num_stages): 

230 if self.strides[i] == 2 or self.downsamples[i - 1]: 

231 whole_downsample_rate *= 2 

232 assert (h % whole_downsample_rate == 0) \ 

233 and (w % whole_downsample_rate == 0),\ 

234 f'The input image size {(h, w)} should be divisible by the whole '\ 

235 f'downsample rate {whole_downsample_rate}, when num_stages is '\ 

236 f'{self.num_stages}, strides is {self.strides}, and downsamples '\ 

237 f'is {self.downsamples}.'
